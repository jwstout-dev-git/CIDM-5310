       4. Show all values for sepal length and sepal height that are within +/-  one Standard Deviation from the mean of those values

SELECT sepal_length, class

FROM jwstout_test.irisdata t

INNER JOIN (

                select

    STD(sepal_length) as stddev,

    AVG(sepal_length) as mean

    FROM jwstout_test.irisdata

) X

ON t.sepal_length >= X.mean - X.stddev

AND

t.sepal_length <= X.mean + X.stddev

ORDER BY sepal_length, class;
